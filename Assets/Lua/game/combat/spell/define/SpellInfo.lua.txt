---@class SpellInfo:Class
---@field spell_id
---@field cooldown_rate
---@field cooldown_remain_duration
---@field level
local SpellInfo = Class("SpellInfo")

function SpellInfo:GetCooldownPct()
  local spellDefinition = global.definitionManager:GetSpellDefinition(self.spell_id)
  local cooldown_duration = spellDefinition.cooldown_duration
  return self.cooldown_remain_duration / (cooldown_duration * self.cooldown_rate)
end



return SpellInfo
