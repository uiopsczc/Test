---@class GuidManager
local GuidManager = Class("GuidManager")

function GuidManager:Init(currentKeyNumber)
	self.keyNumber = currentKeyNumber or 0
end

function GuidManager:NewGuid(id)
	self.keyNumber = self.keyNumber + 1
	return (string.IsNilOrWhiteSpace(id) and "" or tostring(id)) .. IdConst.Rid_Infix .. self.keyNumber
end

return GuidManager
