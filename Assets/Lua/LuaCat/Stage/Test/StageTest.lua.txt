---@class StageTest:StageBase
local StageTest = Class("StageTest", StageBase)

function StageTest:Init()
	StageTest.super.Init(self)
	self.isShowFade = true
	self.isShowLoading = false
	self.stageName = "StageTest"
	self.scenePath = "Assets/Resources/common/ui/scene/StageTestScene.unity"
end

function StageTest:LoadPanels()
	StageTest.super.LoadPanels(self)
	table.insert(self.panelList, global.client.uiManager:CreateChildPanel("UITestPanel", require("luacat.ui.common.UITest.UITestPanel")))
end

function StageTest:Show()
	StageTest.super.Show(self)
	self:HideFade()
end

return StageTest
