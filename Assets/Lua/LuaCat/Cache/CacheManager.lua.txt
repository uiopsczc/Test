---@class CacheManager
local CacheManager = Class("CacheManager")

function CacheManager:Init()
	---@type table<string,Cache>
	self.cacheDict = {}
end

---@return Cache
function CacheManager:GetOrAddCache(key)
	if self.cacheDict[key] == nil then
		self.cacheDict[key] = Cache.New()
	end
	return self.cacheDict[key]
end

function CacheManager:Remove(key)
	if self.cacheDict[key] then
		self.cacheDict[key]:Clear()
	end
	self.cacheDict[key] = nil
end

function CacheManager:IsContainsKey(key)
	if self.cacheDict[key] ~= nil then
		return true
	end
	return false
end

return CacheManager
