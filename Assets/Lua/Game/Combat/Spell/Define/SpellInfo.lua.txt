---@class SpellInfo:Class
---@field spell_id
---@field coolDownRate
---@field coolDownRemainDuration
---@field level
local SpellInfo = Class("SpellInfo")

function SpellInfo:GetCoolDownPct()
	local cfgSpellData = CfgSpell.get_by_id(self.spell_id)
	local coolDownDuration = cfgSpellData.coolDownDuration
	return self.coolDownRemainDuration / (coolDownDuration * self.coolDownRate)
end

return SpellInfo
