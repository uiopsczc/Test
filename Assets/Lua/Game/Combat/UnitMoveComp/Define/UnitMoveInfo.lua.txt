---@class UnitMoveInfo
---@field animationSpeed number
---@field animationName string
---@field speed number
---@field remainDuration number
---@field targetPos Vector3
---@field targetIndexInPath number
---@field path Vector3[]
---@field lookAtUnit Unit
---@field rotateRemainDuration number
---@field endRotation Quaternion
local UnitMoveInfo = Class("UnitMoveInfo")

function UnitMoveInfo:Init()
	self.animationSpeed = -1
	self.animationName = AnimationNameConst.walk
	self.speed = 1
	self.rotateRemainDuration = 0
end

function UnitMoveInfo:IsHasAnimationName()
	return not string.IsNilOrWhiteSpace(self.animationName)
end

return UnitMoveInfo
