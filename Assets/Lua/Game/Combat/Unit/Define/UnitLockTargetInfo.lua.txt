---@class UnitLockTargetInfo:Class
---@field lockTargetUnit Unit
---@field lockTargetPos Vector3
local UnitLockTargetInfo = Class("UnitLockTargetInfo")

function UnitLockTargetInfo:IsHasLockTarget()
	return self.lockTargetUnit or self.lockTargetPos
end

function UnitLockTargetInfo:GetLockTargetPosition()
	if self.lockTargetUnit then
		return self.lockTargetUnit:GetPosition()
	end
	if self.lockTargetPos then
		return self.lockTargetPos
	end
end

return UnitLockTargetInfo
