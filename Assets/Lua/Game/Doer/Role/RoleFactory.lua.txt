---@class RoleFactory:DoerFactory
local RoleFactory = Class("RoleFactory", DoerFactory)

function RoleFactory:Init()
	RoleFactory.super.Init(self)
	self._defaultDoerClassPath = "Game.Doer.Role.Role"
end

function RoleFactory:GetClassPath(id)
	return string.IsNilOrEmpty(self:GetCfgRoleData(id).class_path_lua) and self._defaultDoerClassPath or self:GetCfgRoleData(id).class_path_lua
end

function RoleFactory:GetCfgRoleData(id)
	return CfgRole.get_by_id(id)
end

---@return RoleDBase
function RoleFactory:_NewDBase(id_or_rid)
	return RoleDBase.New(id_or_rid)
end

return RoleFactory
