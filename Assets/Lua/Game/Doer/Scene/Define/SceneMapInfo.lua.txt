---@class SceneMapInfo
local SceneMapInfo = Class("SceneMapInfo")

function SceneMapInfo:Init(grids, projectGrids, offsetPosX, offsetPosY)
	---@type number[][]
	self.grids = grids
	---@type number[][]
	self.projectGrids = projectGrids
	---@type Vector2
	self.offsetPos = Vector2.New(offsetPosX, offsetPosY)
end

return SceneMapInfo
