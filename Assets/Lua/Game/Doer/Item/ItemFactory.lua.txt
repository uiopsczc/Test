---@class ItemFactory:DoerFactory
local ItemFactory = Class("ItemFactory", DoerFactory)

function ItemFactory:Init()
	ItemFactory.super.Init(self)
	self._defaultDoerClassPath = "Game.Doer.Item.Item"
end

function ItemFactory:GetClassPath(id)
	return string.IsNilOrEmpty(self:GetCfgItemData(id).class_path_lua) and self._defaultDoerClassPath or self:GetCfgItemData(id).class_path_lua
end

function ItemFactory:GetCfgItemData(id)
	return CfgItem.get_by_id(id)
end

---@return ItemDBase
function ItemFactory:_NewDBase(id_or_rid)
	return ItemDBase.New(id_or_rid)
end

return ItemFactory
